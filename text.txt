sudo apt update
sudo apt install unzip –y
sudo unzip gradle-8.7-bin.zip -d /opt/
sudo nano /etc/profile.d/gradle.sh
sudo chmod +x /etc/profile.d/gradle.sh && source /etc/profile.d/gradle.sh

pipeline {
    agent any

    stages {

        stage('Checkout') {
            steps {
                // Check out code from Git repository
                git url: 'https://github.com/NiharikaMK/Marven-Project', branch: 'main'
            }
        }

        stage('Build') {
            steps {
                // Run Maven build
                sh 'mvn clean package'
            }
        }

        stage('Test') {
            steps {
                // Optionally, separate test execution if needed
                sh 'mvn test'
            }
        }
    }

    post {

        always {
            // Archive test reports
            junit '**/target/surefire-reports/*.xml'
        }

        success {
            echo 'Build and tests succeeded!'
        }

        failure {
            echo 'Build or tests failed.'
        }
    }
}
